x_1 - x_2 +x_3 = -2 \\
2x_1 + x_2 - 2x_3 = 6 \\
x_1 + 2x_2 + 3x_3 = 2\\
\end{cases} \\
D = \begin{vmatrix}
1  &  -1  &  1   \\
2  &   1  & -2   \\
1  &   2  &  3   \\
\end{vmatrix}  = 18 \not = 0,

D_1 = \begin{vmatrix}
-2 &  -1  &  1   \\
6  &   1  & -2   \\
2  &   2  &  3   \\
\end{vmatrix}  = 18, \\ 

D_2 = \begin{vmatrix}
1 & -2 &    1   \\
2 & 6  &   -2   \\
1 & 2  &    3   \\
\end{vmatrix}  = 36,

D_3 = \begin{vmatrix}
1 &  -1  & -2  \\
2 &   1  & 6   \\
1 &   2  & 2   \\
\end{vmatrix}  = -18 \\
x_1 = \frac{D_1}{D} = \frac{18}{18} = 1 ;
x_2 = \frac{D_2}{D} = \frac{36}{18} = 2 ;
x_3 = \frac{D_3}{D} = \frac{-18}{18} = -1 .
