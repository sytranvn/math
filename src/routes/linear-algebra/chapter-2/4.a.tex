\end{cases} \\

\overline{A} = (A|B) = \begin{pmatrix}
\begin{array}{c c c | c}
m & 1 & 1 & 1   \\
1 & m & 1 & m   \\
1 & 1 & m & m^2 \\
\end{array}
\end{pmatrix} \\

\xrightarrow[]{(1) \sim (2)}
\begin{pmatrix}
\begin{array}{c c c | c}

\end{array}
\end{pmatrix}

\xrightarrow[(3):=(3)-(1)]{(2):=(2) -m(1)}
\begin{pmatrix}
\begin{array}{c c c | c}

\end{array}
\end{pmatrix} \\

\xrightarrow[]{(2) \sim (3)}
\begin{pmatrix}
\begin{array}{c c c | c}

\end{array}
\end{pmatrix} \\


\xrightarrow[]{(3):=(3)-(1+m)(2)}
\begin{pmatrix}
\begin{array}{c c c | c}

\end{array}
\end{pmatrix} \\
m=-2: rank A = 2 < rank \overline{A} = 3. \text{No solution} \\
m=1: rank A = 1 = rank \overline{A} = 1 < 3 \text{. Infinite solutions} \\
m=0: rank A = rank \overline{A} = 3. \\
